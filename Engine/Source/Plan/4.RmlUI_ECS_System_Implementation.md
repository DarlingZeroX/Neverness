```cpp
/// @brief RmlUI 绘制项——纯数据，无 Rml SDK 依赖。
struct RmlDrawItem
{
    uint32_t entityId = 0;
    NNGuid assetGuid{};
    int32_t sortOrder = 0;
    NNRmlUIVisibility visibility = NNRmlUIVisibility::Both;
    bool enabled = false;
};

/// @brief RmlUI ECS 系统——只负责构建 DrawList。
class NNRmlUISystem : public ISceneSystem
{
public:
    NNSceneTickGroup TickGroup() const noexcept override;  // EarlyUpdate
    void Tick(NNRuntimeScene& scene, float dt) noexcept override;

    const std::vector<RmlDrawItem>& GetDrawList() const;

private:
    void SyncDocuments(NNRuntimeScene& scene);

    // EntityId → 组件快照（用于 diff）
    struct EntrySnapshot
    {
        NNGuid assetGuid{};
        int32_t sortOrder = 0;
        NNRmlUIVisibility visibility = NNRmlUIVisibility::Both;
        NNRmlUIDocumentFlags flags{};
        bool alive = false;
    };

    std::unordered_map<uint32_t, EntrySnapshot> m_Snapshots;
    std::vector<RmlDrawItem> m_DrawList;
    bool m_DrawListDirty = true;
};
```

**Tick 实现**：

```cpp
void NNRmlUISystem::Tick(NNRuntimeScene& scene, float dt) noexcept
{
    SyncDocuments(scene);
}

void NNRmlUISystem::SyncDocuments(NNRuntimeScene& scene)
{
    // 1. 标记所有快照为 not alive
    for (auto& [id, snap] : m_Snapshots)
        snap.alive = false;

    // 2. 扫描 ECS，更新快照
    scene.Query<NNRmlUIDocumentComponent>().Each(
        [&](NNEntity entity, NNRmlUIDocumentComponent& comp)
        {
            uint32_t id = entity.Value;
            auto& snap = m_Snapshots[id];
            bool changed = (snap.assetGuid != comp.DocumentAsset ||
                           snap.sortOrder != comp.SortOrder ||
                           snap.visibility != comp.Visibility ||
                           snap.flags != comp.Flags);
            if (changed) m_DrawListDirty = true;

            snap.assetGuid = comp.DocumentAsset;
            snap.sortOrder = comp.SortOrder;
            snap.visibility = comp.Visibility;
            snap.flags = comp.Flags;
            snap.alive = true;
        });

    // 3. 移除已销毁实体
    for (auto it = m_Snapshots.begin(); it != m_Snapshots.end(); )
    {
        if (!it->second.alive)
        {
            m_DrawListDirty = true;
            it = m_Snapshots.erase(it);
        }
        else
            ++it;
    }

    // 4. 重建 DrawList（仅在脏时）
    if (m_DrawListDirty)
    {
        m_DrawList.clear();
        for (const auto& [id, snap] : m_Snapshots)
        {
            if (snap.assetGuid.High == 0 && snap.assetGuid.Low == 0)
                continue;

            m_DrawList.push_back({
                .entityId = id,
                .assetGuid = snap.assetGuid,
                .sortOrder = snap.sortOrder,
                .visibility = snap.visibility,
                .enabled = HasFlag(snap.flags, NNRmlUIDocumentFlags::AutoLoad),
            });
        }

        // 按 SortOrder 排序
        std::sort(m_DrawList.begin(), m_DrawList.end(),
            [](const RmlDrawItem& a, const RmlDrawItem& b)
            { return a.sortOrder < b.sortOrder; });

        m_DrawListDirty = false;
    }
}
```

```cpp
/// @brief 文档运行时状态机。
enum class RmlDocState : uint8_t
{
    Loading,    // 正在加载（路径解析中）
    Ready,      // 已加载，可渲染
    Hidden,     // 已加载但不可见
    Destroyed,  // 标记销毁，下一帧清理
};

/// @brief 文档运行时实例（Renderer 内部管理）。
struct RmlDocumentRuntime
{
    Rml::ElementDocument* doc = nullptr;
    RmlDocState state = RmlDocState::Loading;
    NNGuid assetGuid{};
};

/// @brief RmlUI 渲染器——持有 Context，管理文档实例，从 DrawList 渲染。
class RmlUIRenderer
{
public:
    bool Initialize(uint32_t viewportWidth, uint32_t viewportHeight);
    void Shutdown();
    void SetViewport(uint32_t width, uint32_t height);

    /// @brief 渲染。Renderer 内部负责：同步 DrawList → 更新 → 渲染。
    void Render(const std::vector<RmlDrawItem>& drawList,
                NNRmlUIVisibility viewMode);

    void ProcessInput(const NNRmlUIInputEvent& event);
    void SetAssetResolver(IAssetResolver* resolver);

private:
    /// @brief 同步 DrawList → 内部文档实例。
    /// 新 entityId → GetOrCreateDocument
    /// 不在 list → HideDocument
    /// 已销毁 → UnloadDocument
    void SyncDrawList(const std::vector<RmlDrawItem>& drawList);

    Rml::ElementDocument* GetOrCreateDocument(uint32_t entityId, NNGuid assetGuid);
    void HideDocument(uint32_t entityId);
    void UnloadDocument(uint32_t entityId);

    Rml::Context* m_Context = nullptr;
    RenderInterface_GL3* m_RenderInterface = nullptr;
    SystemInterface_SDL* m_SystemInterface = nullptr;
    UIFileInterfaceVFS* m_FileInterface = nullptr;
    IAssetResolver* m_AssetResolver = nullptr;

    // EntityId → 文档运行时实例

    uint32_t m_ViewportHeight = 0;
```

**Render 实现**：

```cpp
void RmlUIRenderer::Render(const std::vector<RmlDrawItem>& drawList,
                           NNRmlUIVisibility viewMode)
{
    if (!m_Initialized) return;

    // 1. 同步 DrawList（加载/卸载/隐藏文档）
    SyncDrawList(drawList);

    // 2. 更新 Context
    m_Context->Update();

    // 3. 渲染
    m_RenderInterface->BeginFrame();

    for (const auto& item : drawList)
    {
        // Visibility 过滤
        if (item.visibility != NNRmlUIVisibility::Both &&
            item.visibility != viewMode)
            continue;

        if (!item.enabled) continue;

        auto it = m_Documents.find(item.entityId);
        if (it != m_Documents.end() && it->second.state == RmlDocState::Ready)
        {
            it->second.doc->UpdateDocument();
            it->second.doc->Render();
        }
    }

    m_RenderInterface->EndFrame();
}
```

**SyncDrawList 实现**：

```cpp
void RmlUIRenderer::SyncDrawList(const std::vector<RmlDrawItem>& drawList)
{
    // 收集 active entityId 集合
    std::unordered_set<uint32_t> activeIds;
    for (const auto& item : drawList)
        activeIds.insert(item.entityId);

    // 卸载不在 list 中的文档
    for (auto it = m_Documents.begin(); it != m_Documents.end(); )
    {
        if (!activeIds.contains(it->first))
        {
            UnloadDocument(it->first);
            it = m_Documents.erase(it);
        }
        else
            ++it;
    }

    // 加载新文档（lazy）
    for (const auto& item : drawList)
    {
        if (m_Documents.contains(item.entityId))
            continue;

        GetOrCreateDocument(item.entityId, item.assetGuid);
    }
}

Rml::ElementDocument* RmlUIRenderer::GetOrCreateDocument(
    uint32_t entityId, NNGuid assetGuid)
{
    // 已存在
    auto it = m_Documents.find(entityId);
    if (it != m_Documents.end())
        return it->second.doc;

    // 解析路径
    char path[512];
    if (!m_AssetResolver || !m_AssetResolver->Resolve(assetGuid, path, sizeof(path)))
    {
        m_Documents[entityId] = { nullptr, RmlDocState::Loading, assetGuid };
        return nullptr;
    }

    // 加载文档
    auto* doc = m_Context->LoadDocument(path);
    if (doc)
    {
        doc->Show();
        m_Documents[entityId] = { doc, RmlDocState::Ready, assetGuid };
    }
    else
    {
        m_Documents[entityId] = { nullptr, RmlDocState::Destroyed, assetGuid };
    }

    return doc;
}

void RmlUIRenderer::HideDocument(uint32_t entityId)
{
    auto it = m_Documents.find(entityId);
    if (it != m_Documents.end() && it->second.doc)
    {
        it->second.doc->Hide();
        it->second.state = RmlDocState::Hidden;
    }
}

void RmlUIRenderer::UnloadDocument(uint32_t entityId)
{
    auto it = m_Documents.find(entityId);
    if (it != m_Documents.end())
    {
        if (it->second.doc)
            it->second.doc->Close();
        it->second.state = RmlDocState::Destroyed;
    }
}
```
