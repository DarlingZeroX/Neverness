/// @brief RmlUI 绘制项——纯数据，无 Rml SDK 依赖。
struct RmlDrawItem
{
    NNEntity entity{};          // 带 generation 的实体句柄（防复用）
    NNGuid assetGuid{};
    int32_t sortOrder = 0;
    NNRmlUIViewTarget viewTarget = NNRmlUIViewTarget::Both;
};

/// @brief RmlUI ECS 系统——只负责构建 DrawList。
class NNRmlUISystem : public ISceneSystem
{
public:
    NNSceneTickGroup TickGroup() const noexcept override;  // EarlyUpdate
    void Tick(NNRuntimeScene& scene, float dt) noexcept override;

    const std::vector<RmlDrawItem>& GetDrawList() const;

private:
    void SyncDocuments(NNRuntimeScene& scene);

    // EntityId → 组件快照（用于 diff）
    struct EntrySnapshot
    {
        NNGuid assetGuid{};
        int32_t sortOrder = 0;
        NNRmlUIViewTarget viewTarget = NNRmlUIViewTarget::Both;
        NNRmlUIDocumentFlags flags{};
        bool alive = false;
    };

    std::unordered_map<uint32_t, EntrySnapshot> m_Snapshots;
    std::vector<RmlDrawItem> m_DrawList;
    bool m_DrawListDirty = true;
};

**Tick 实现**：

```cpp
void NNRmlUISystem::Tick(NNRuntimeScene& scene, float dt) noexcept
{
    SyncDocuments(scene);
}

void NNRmlUISystem::SyncDocuments(NNRuntimeScene& scene)
{
    // 1. 标记所有快照为 not alive
    for (auto& [id, snap] : m_Snapshots)
        snap.alive = false;

    // 2. 扫描 ECS，更新快照
    scene.Query<NNRmlUIDocumentComponent>().Each(
        [&](NNEntity entity, NNRmlUIDocumentComponent& comp)
        {
            uint32_t id = entity.Value;
            auto& snap = m_Snapshots[id];
            bool changed = (snap.assetGuid != comp.DocumentAsset ||
                           snap.sortOrder != comp.SortOrder ||
                           snap.viewTarget != comp.ViewTarget ||
                           snap.flags != comp.Flags);
            if (changed) m_DrawListDirty = true;

            snap.assetGuid = comp.DocumentAsset;
            snap.sortOrder = comp.SortOrder;
            snap.viewTarget = comp.ViewTarget;
            snap.flags = comp.Flags;
            snap.alive = true;
        });

    // 3. 移除已销毁实体
    for (auto it = m_Snapshots.begin(); it != m_Snapshots.end(); )
    {
        if (!it->second.alive)
        {
            m_DrawListDirty = true;
            it = m_Snapshots.erase(it);
        }
        else
            ++it;
    }

    // 4. 重建 DrawList（仅在脏时）
    if (m_DrawListDirty)
    {
        m_DrawList.clear();
        for (const auto& [id, snap] : m_Snapshots)
        {
            if (snap.assetGuid.High == 0 && snap.assetGuid.Low == 0)
                continue;

            m_DrawList.push_back({
                .entityId = id,
                .assetGuid = snap.assetGuid,
                .sortOrder = snap.sortOrder,
                .viewTarget = snap.viewTarget,
                .enabled = HasFlag(snap.flags, NNRmlUIDocumentFlags::AutoLoad),
            });
        }

        // 按 SortOrder 排序
        std::sort(m_DrawList.begin(), m_DrawList.end(),
            [](const RmlDrawItem& a, const RmlDrawItem& b)
            { return a.sortOrder < b.sortOrder; });

        m_DrawListDirty = false;
    }
}
```

## Step 3: RmlUIRenderer（v4 修正版）

**⚠️ 关键修正（基于架构审查反馈）：**
1. **Sync() 从 Render() 中拆出** — 文档加载/卸载在 Update 阶段，Render 纯渲染
2. **EntityKey 用 NNEntity（带 generation）** — 防止 Entity 复用导致旧 Document 绑定新 Entity
3. **移除 Loading 状态** — LoadDocument 是同步的，Loading 只存在几微秒，无意义
4. **Visibility → ViewTarget** — 避免与 Visible/Hidden 混淆
5. **AutoLoad 只影响初始加载** — 运行时启用/禁用用单独的 Enabled 状态

/// @brief 文档运行时状态（简化，无 Loading）。
enum class RmlDocState : uint8_t
{
    Ready,      // 已加载，可渲染
    Hidden,     // 已加载但不可见
    Failed,     // 加载失败
};

/// @brief 文档运行时实例（Renderer 内部管理）。
struct RmlDocumentRuntime
{
    Rml::ElementDocument* doc = nullptr;
    RmlDocState state = RmlDocState::Ready;
    NNGuid assetGuid{};
};

/// @brief RmlUI 渲染器——持有 Context，管理文档实例。
///
/// 生命周期分离：
/// - Sync()  — 在 Update 阶段调用，同步 DrawList → 加载/卸载文档
/// - Update() — 更新 Context（Layout + 动画）
/// - Render() — 纯渲染，不加载/卸载任何资源
class RmlUIRenderer
{
public:
    bool Initialize(uint32_t viewportWidth, uint32_t viewportHeight);
    void Shutdown();
    void SetViewport(uint32_t width, uint32_t height);

    /// @brief 同步 DrawList（Update 阶段调用）。
    /// 加载新文档、卸载不再需要的文档。
    /// 不在 Render 阶段调用。
    void Sync(const std::vector<RmlDrawItem>& drawList);

    /// @brief 更新 Context（Update 阶段调用）。
    void Update();

    /// @brief 纯渲染（Render 阶段调用）。
    /// 不加载/卸载任何资源。
    void Render(const std::vector<RmlDrawItem>& drawList,
                NNRmlUIViewTarget viewTarget);

    void ProcessInput(const NNRmlUIInputEvent& event);
    void SetAssetResolver(IAssetResolver* resolver);

private:
    Rml::ElementDocument* LoadDocument(NNGuid assetGuid);
    void UnloadDocument(NNEntity entity);

    Rml::Context* m_Context = nullptr;
    RenderInterface_GL3* m_RenderInterface = nullptr;
    SystemInterface_SDL* m_SystemInterface = nullptr;
    UIFileInterfaceVFS* m_FileInterface = nullptr;
    IAssetResolver* m_AssetResolver = nullptr;

    // NNEntity（带 generation）→ 文档运行时实例

    uint32_t m_ViewportHeight = 0;

**Sync 实现（Update 阶段，不在 Render 中）**：

```cpp
void RmlUIRenderer::Sync(const std::vector<RmlDrawItem>& drawList)
{
    // 收集 active entity 集合
    std::unordered_set<NNEntity, NNEntityHash> activeEntities;
    for (const auto& item : drawList)
        activeEntities.insert(item.entity);

    // 卸载不在 list 中的文档
    for (auto it = m_Documents.begin(); it != m_Documents.end(); )
    {
        if (!activeEntities.contains(it->first))
        {
            UnloadDocument(it->first);
            it = m_Documents.erase(it);
        }
        else
            ++it;
    }

    // 加载新文档
    for (const auto& item : drawList)
    {
        if (m_Documents.contains(item.entity))
            continue;

        auto* doc = LoadDocument(item.assetGuid);
        if (doc)
        {
            doc->Show();
            m_Documents[item.entity] = { doc, RmlDocState::Ready, item.assetGuid };
        }
        else
        {
            m_Documents[item.entity] = { nullptr, RmlDocState::Failed, item.assetGuid };
        }
    }
}

Rml::ElementDocument* RmlUIRenderer::LoadDocument(NNGuid assetGuid)
{
    char path[512];
    if (!m_AssetResolver || !m_AssetResolver->Resolve(assetGuid, path, sizeof(path)))
        return nullptr;
    return m_Context->LoadDocument(path);
}

void RmlUIRenderer::UnloadDocument(NNEntity entity)
{
    auto it = m_Documents.find(entity);
    if (it != m_Documents.end() && it->second.doc)
        it->second.doc->Close();
}
```

```cpp
void RmlUIRenderer::Render(const std::vector<RmlDrawItem>& drawList,
                           NNRmlUIViewTarget viewTarget)
{
    if (!m_Initialized) return;

    m_RenderInterface->BeginFrame();

    for (const auto& item : drawList)
    {
        // ViewTarget 过滤
        if (item.viewTarget != NNRmlUIViewTarget::Both &&
            item.viewTarget != viewTarget)
            continue;

        auto it = m_Documents.find(item.entity);
        if (it != m_Documents.end() && it->second.state == RmlDocState::Ready)
        {
            it->second.doc->UpdateDocument();
            it->second.doc->Render();
        }
    }

    m_RenderInterface->EndFrame();
}
```

## ⚠️ 架构问题：RmlUI Renderer 在 Editor 中的 FBO 冲突

**问题**：`RmlUIRenderer::Render()` 调用 `RenderInterface_GL3::BeginFrame()` + `EndFrame()`，其中 `EndFrame()` 会：
1. 解析 MSAA（blit 内部 FBO → 后处理 FBO）
2. 绑定 framebuffer 0（屏幕）
3. 渲染全屏四边形
4. 恢复 GL state

但在 Editor 中，`RenderSceneToTexture` 是从 ImGui `OnGUI()` 内部调用的。此时 ImGui 已在渲染帧中，RmlUI 的 `EndFrame()` 破坏 ImGui 的 GL 状态 → SEH 崩溃。

**解决方案**：需要修改 RmlUI 渲染流程，使其渲染到独立 FBO 并暴露纹理 ID，而不是 blit 到屏幕。

**临时方案**：已注释掉 `Update()` + `Render()` 调用，仅保留 `Sync()`（文档加载/卸载），避免崩溃。
